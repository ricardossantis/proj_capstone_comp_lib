## Install

```bash
npm install --save proj-capstone-lib
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'proj-capstone-lib'
import 'proj-capstone-lib/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
